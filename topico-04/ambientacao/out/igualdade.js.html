function iguais(a,b){
    let propertiesOfA = Object.keys(a);
    let propertiesOfB = Object.keys(b);

    if(propertiesOfA.length !== propertiesOfB.length){
        return false;
    }
    return !propertiesOfA.some( e => a[e] !== b[e]);
}
